import 'dart:convert';
import 'dart:io';
import 'package:flutter/foundation.dart';
import 'package:path_provider/path_provider.dart';
import '../models/purchase.dart';

class PurchaseService {
  Future<String> _getFilePath() async {
    final directory = await getApplicationDocumentsDirectory();
    return '${directory.path}/purchases.json';
  }

  Future<List<Purchase>> getPurchases() async {
    try {
      final filePath = await _getFilePath();
      final file = File(filePath);
      if (await file.exists()) {
        final jsonString = await file.readAsString();
        final List<dynamic> jsonList = jsonDecode(jsonString);
        return jsonList.map((json) => Purchase.fromJson(json)).toList();
      }
      return [];
    } catch (e) {
      debugPrint("‚ö†Ô∏è Fehler beim Laden der Eink√§ufe: $e");
      return [];
    }
  }

  Future<void> addPurchase(Purchase purchase) async {
    try {
      final filePath = await _getFilePath();
      final file = File(filePath);
      List<Purchase> purchases = await getPurchases();
      purchases.add(purchase);
      final jsonString = jsonEncode(purchases.map((p) => p.toJson()).toList());
      debugPrint("üìÑ Schreibe in purchases.json: $jsonString");
      await file.writeAsString(jsonString);
      debugPrint(
          "‚úÖ Einkauf hinzugef√ºgt: ${purchase.itemName}, ${purchase.price} CHF");
    } catch (e) {
      debugPrint("‚ö†Ô∏è Fehler beim Speichern des Einkaufs: $e");
    }
  }
}
